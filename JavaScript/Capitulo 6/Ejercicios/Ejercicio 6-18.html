Alba García José Iván

Escriba una secuencia de comandos que reciba tres enteros del usuario y muestre la suma, promedio, producto, menor y mayor de los números en un diálogo alert.

var n1, n2, n3;

n1 = window.prompt("Primer entero: ", 0);

n2 = window.prompt("Segundo entero: ", 0);

n3 = window.prompt("Tercer entero: ", 0);

(n1 =="" || n1 == null) ? n1 = 0 : null;

(n2 =="" || n2 == null) ? n2 = 0 : null;

(n3 =="" || n3 == null) ? n3 = 0 : null;

n1 = parseInt(n1);

n2 = parseInt(n2);

n3 = parseInt(n3);

document.write("Suma = " + (n1 + n2 + n3) + "<br>");

document.write("Promedio = " + ((n1 + n2 + n3)/3) + "<br>");

document.write("Producto = " + (n1 * n2 * n3) + "<br>");

((n1 > n2) && (n1 > n3)) ? document.write(n1 + " es el número mayor.<br>") : ((n2 > n1) && (n2 > n3)) ? document.write(n2 + " es el número mayor.<br>") : document.write(n3 + " es el número mayor.<br>");

((n1 < n2) && (n1 < n3)) ? document.write(n1 + " es el número menor.<br>") : ((n2 < n1) && (n2 < n3)) ? document.write(n2 + " es el número menor.<br>") : document.write(n3 + " es el número menor.<br>");
