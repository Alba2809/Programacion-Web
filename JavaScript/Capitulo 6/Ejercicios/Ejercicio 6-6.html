Use las instrucciones que escribió en el ejercicio 6.5 para escribir una secuencia de comandos completa que calcule e imprima el productos de tres enteros.

var x, y, z, resultado;
var xVal, yVal, zVal;
xVal = window.prompt("Introduzca el primer valor:");
yVal = window.prompt("Introduzca el segundo valor:");
zVal = window.prompt("Introduzca el tercer valor:");
x = parseInt(xVal);
y = parseInt(yVal);
z = parseInt(zVal);
resultado = x * y * z;
document.writeln("<h3>El resultado es: " + resultado + "</h3>");
