Escriba una secuencia de comandos que calcule los cuadrados y cubos de los números del 0 al 5 y produzca texto de HTML5 que muestre los valores resultantes en un formato de tabla de HTML5, como se muestra a continuación. [Nota: esta secuencia de comandos no requiere entrada del usuario].

document.write("<table border='1' style='text-align: center; width: 350px;'>");
document.write("<thead>");
document.write("<tr>");
document.write("<th>número</th><th>cuadrado</th><th>cubo</th>");
document.write("</tr>");
document.write("</thead>");
document.write("<tbody>");

for (let index = 0; index < 6; index++) {
document.write("<tr>");
document.write("<td>" + index + "</td>");
document.write("<td>" + (index * index) + "</td>");
document.write("<td>" + (index * index * index) + "</td>");
document.write("</tr>");
}

document.write("</tbody>");
document.write("</table>");
