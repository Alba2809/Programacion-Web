Escriba una secuencia de comandos que reciba tres números, determine y produzca marcado que muestre la cantidad de números negativos, positivos y ceros introducidos.

var n = [], negativos = 0, positivos = 0, ceros = 0;

n[0] = window.prompt("Primer número:", 0);
n[1] = window.prompt("Segundo número:", 0);
n[2] = window.prompt("Tercer número:", 0);

(n[0] == "" || n[0] == null) ? n[0] = 0 : null;
(n[1] == "" || n[1] == null) ? n[1] = 0 : null;
(n[2] == "" || n[2] == null) ? n[2] = 0 : null;

n[0] = parseInt(n[0]);
n[1] = parseInt(n[1]);
n[2] = parseInt(n[2]);

for (var index = 0; index < 3; index++) {
(n[index] < 0) ? negativos++ : null;
(n[index] == 0) ? ceros++ : null;
(n[index] > 0) ? positivos++ : null;

}

document.write("Negativos: " + negativos + "<br>");
document.write("Positivos: " + positivos + "<br>");
document.write("Ceros: " + ceros);
